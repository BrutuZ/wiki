## DivXATope


```
rss:
  url: http://divxatope.com/feeds.xml
series:
  settings:
    normales:
      ep_regexp:
        - Cap.(\d+)(\d\d)
      quality: hdtv <720p
      path: /media/269c2dba-2000-495f-9bd8-e09c36b16f4e/series/{{series_name}}
  normales:
    - Alli abajo
    - Arrow
    - Westworld
regexp:
  reject:
    - V.O.:
        from: title
```
